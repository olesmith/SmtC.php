\setboolean{WithSolutionOnly}{false}  
\newcommand{\ListItemLong}[3]
{
    \IfFileExists{#1/#2.Solution.tex}
    {
       \begin{addmargin}[1em]{2em}% 1em left, 2em right
          \textbf{#3:}
          \input{#1/#2.tex}
          
          \ListItemTitle             
          \begin{addmargin}[1em]{2em}% 1em left, 2em right
              \input{#1/#2.Solution.tex}
          \end{addmargin}

        \vspace{0.35cm}
             
          \ListSolutionTrailer
        \end{addmargin}
        \vspace{0.35cm}
      }
      {
         \ifthenelse{\boolean{WithSolutionOnly}}
         {}
         {
             \begin{addmargin}[1em]{2em}% 1em left, 2em right
                \textbf{#3:}
                \input{#1/#2.tex}
              \end{addmargin}
         }          
         \vspace{0.35cm}
      }
}
